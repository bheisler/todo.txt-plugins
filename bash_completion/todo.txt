function _todo.sh() {
    our_cmd=${COMP_WORDS[0]}
    cmd=${COMP_CWORD}
    COMPREPLY=()
    COMP_WORDBREAKS=${COMP_WORDBREAKS//:}
    word="${COMP_WORDS[COMP_CWORD]}"
    COMMANDS="\
        add a addto addm append app archive command del  \
        rm depri dp do help list ls listaddons listall lsa listcon  \
        lsc listfile lf listpri lsp listproj lsprj move \
        mv prepend prep pri p replace report shorthelp"
    if ((cmd==1)); then
        # Help me pick a todo.sh action
        COMPREPLY=($(compgen -W "$(eval $our_cmd listaddons ; echo $COMMANDS)" -- "${word}"));
    else
        # help me pick a project
        case ${COMP_WORDS[1]} in
            # help me pick a note file
            "na"|"notesarchive"|"nc"|"notescat"|"ne"|"notesedit")
                COMPREPLY=($(compgen -W "$(eval $our_cmd lsn | tr -d '\r')" -- "${word}"));
            ;;
            # help me pick a context or project
            "a"|"add"|"addto"|"ls"|"list")
                COMPREPLY=($(compgen -W "$(eval $our_cmd lsp | tr -d '\r' | sed 's/^/+/g') $(eval $our_cmd lsc | tr -d '\r') $(eval $our_cmd lsn | tr -d '\r')" -- "${word}"));
            ;;
            # help me pick a context of project if an item number is already set
            "app"|"append"|"prep"|"prepend"|"replace")
                if [ `echo ${COMP_WORDS[1]} | grep -c "[0-9]"` = 1 ] ; then
                    COMPREPLY=($(compgen -W "$(eval $our_cmd lsp | tr -d '\r' | sed 's/^/+/g') $(eval $our_cmd lsc | tr -d '\r')" -- "${word}"));
                fi
            ;;
            # Find all  option include files
            *)
            for action in $(eval $our_cmd listaddons) ; do
                if [ $(stat /usr/share/todo.txt/bash_completion.d/${action}/${action}.sh > /dev/null 2>&1 ; echo $?) -eq 0 ]; then
                    . /usr/share/todo.txt/bash_completion.d/${action}/${action}.sh
                fi
            done
            ;;
        esac
    fi
}
# add any aliases for todo.sh after _todo.sh
complete -F _todo.sh todo.sh t tt tw tp
